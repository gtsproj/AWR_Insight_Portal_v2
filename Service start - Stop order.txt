net stop AWRPortal

Start order (foundation first):

net start Grafana
net start AWRPortal
net start AWRWatcher
net start SARWatcher
net start AWRQueueProcessor

net stop SARWatcher
net stop AWRWatcher
net stop AWRPortal
net stop Grafana

sc query AWRQueueProcessor

for %s in (AWRPortal AWRWatcher SARWatcher AWRQueueProcessor Grafana) do @sc query %s | findstr "SERVICE_NAME STATE"
